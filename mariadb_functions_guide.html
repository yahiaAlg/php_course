üîç Search & Pattern Matching Functions

Search Functions
Function Use Case Example
LIKE Pattern matching with wildcards
SELECT * FROM customer WHERE name LIKE '%Smith%'
REGEXP Regular expression matching
SELECT * FROM product WHERE name REGEXP '^(Smart|Wireless)'
MATCH...AGAINST Full-text search
SELECT * FROM product WHERE MATCH(name) AGAINST('wireless')
SOUNDEX Phonetic matching
SELECT * FROM customer WHERE SOUNDEX(name) = SOUNDEX('Jon')
IN Multiple value matching
SELECT * FROM order WHERE status IN ('Pending', 'Processing')
BETWEEN Range filtering
SELECT * FROM product WHERE price BETWEEN 100 AND 500
EXISTS Subquery existence check
SELECT * FROM customer WHERE EXISTS (SELECT 1 FROM order WHERE customer_id = customer.id)

üìù String Functions

String Functions
Function Use Case Example
CONCAT Concatenate strings
SELECT CONCAT(name, ' - ', reference) FROM product
LENGTH String length in bytes
SELECT LENGTH(name) FROM customer
CHAR_LENGTH Character count
SELECT CHAR_LENGTH(name) FROM customer
SUBSTRING Extract substring
SELECT SUBSTRING(name, 1, 10) FROM product
LEFT/RIGHT Extract from left/right
SELECT LEFT(name, 5), RIGHT(phone, 4) FROM customer

SELECT UPPER(name), LOWER(email) FROM customer

SELECT TRIM(name) FROM customer
REPLACE Replace text
SELECT REPLACE(phone, '+1-', '') FROM customer
SUBSTRING_INDEX Split by delimiter
SELECT SUBSTRING_INDEX(email, '@', 1) AS username

ü™ü Window Functions

Window Functions
Function Use Case Example
ROW_NUMBER Sequential numbering
SELECT ROW_NUMBER() OVER (ORDER BY price DESC) FROM product
RANK Ranking with gaps
SELECT RANK() OVER (ORDER BY price DESC) FROM product
DENSE_RANK Ranking without gaps
SELECT DENSE_RANK() OVER (ORDER BY price DESC) FROM product
NTILE Divide into buckets
SELECT NTILE(4) OVER (ORDER BY price) AS quartile
LAG/LEAD Previous/next row value
SELECT LAG(price, 1) OVER (ORDER BY created_at)
FIRST_VALUE First value in partition
SELECT FIRST_VALUE(price) OVER (PARTITION BY category ORDER BY price)
LAST_VALUE Last value in partition
SELECT LAST_VALUE(price) OVER (PARTITION BY category ORDER BY price)
PERCENT_RANK Percentile ranking
SELECT PERCENT_RANK() OVER (ORDER BY price)

üîÄ Conditional Functions

Conditional Functions
Function Use Case Example
CASE Conditional logic
SELECT CASE WHEN price > 500 THEN 'Expensive' ELSE 'Affordable' END
IF Simple condition
SELECT IF(discount = 1, 'On Sale', 'Regular') FROM product
IFNULL Handle NULL values
SELECT IFNULL(phone, 'No Phone') FROM customer
NULLIF Return NULL if equal
SELECT NULLIF(discount, 0) FROM product
COALESCE First non-NULL value
SELECT COALESCE(phone, email, 'No Contact') FROM customer
